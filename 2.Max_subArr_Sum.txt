Q.Find the Subarray with Largest Sum ?

def max_subarray(nums):
    max_sum = float('-inf')
    current_sum = 0
    start = 0
    end = 0
    temp_start = 0

    for i in range(len(nums)):
        if current_sum <= 0:
            current_sum = nums[i]
            temp_start = i
        else:
            current_sum += nums[i]

        if current_sum > max_sum:
            max_sum = current_sum
            start = temp_start
            end = i

    return nums[start:end+1], max_sum

# Example usage
nums = [-2,-1,-3,4,-1,2,1,-5,4]
subarray, max_sum = max_subarray(nums)

print(f"The subarray with the largest sum is: {subarray}")
print(f"The sum of this subarray is: {max_sum}")
